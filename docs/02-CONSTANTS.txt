=== MADNESS SCALING FORMULAS ===
getMadnessAmount():
  score < 150k: base = 0
  150k-800k: base = (score-150k)/650k * 0.5
  800k+: base = 0.5 + min(0.5, (score-800k)/600k)
  return min(1, base + horrorBoost)

getMadnessPhysicsScale():
  0-50% madness: 1.0 -> 1.15
  50-100%: 1.15 -> 1.7

getAbyssSaturation():
  0-10% madness: 0 (pitch black)
  10-70% madness: ramp 0->1
  70-100%: 1 (fully vivid)
